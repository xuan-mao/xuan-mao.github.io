Alist + aria2 + OneDrive

install aria2

pacman -S aria2

Download alist

aria2c https://github.com/alist-org/alist/releases/download/v3.35.0/alist-linux-amd64.tar.gz

unzip the downloaded file

tar -zxvf alist-linux-amd64.tar.gz

write a systemd service as this

# .config/systemd/user/alist.service

# alist.service
[Unit]
Description=alist server
After=network.target

[Service]
Type=simple
WorkingDirectory=/home/aunomax/opt
ExecStart=/home/aunomax/opt/alist server
StandardOutput=journal
Restart=on-failure

[Install]
WantedBy=default.target

enable alist.service

systemctl --user enable alist.service
systemctl --user start alist.service
systemctl --user status alist.service

read this add storage OneDrive

aria2

mkdir ~/config/aria2
touch ~/config/aria2.conf
touch ~/config/aria2.session
touch ~/config/aria2.log
chmod 666 ~/config/aria2.log ~/config/aria2.session

edit configuration file as this

helix ~/.aria2/aria2.conf

add a systemd service as this

helix ~/config/systemd/user/aria2.service
